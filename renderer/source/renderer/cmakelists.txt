project(Renderer)
file(GLOB_RECURSE SOURCES_RENDERER CONFIGURE_DEPENDS
    "${CMAKE_CURRENT_SOURCE_DIR}/*.cpp"
    "${CMAKE_CURRENT_SOURCE_DIR}/*.h"
    "${CMAKE_CURRENT_SOURCE_DIR}/*.inl"
)
add_library(Renderer SHARED "${SOURCES_RENDERER}")
target_include_directories(Renderer PRIVATE "${CMAKE_CURRENT_SOURCE_DIR}")
target_include_directories(Renderer PRIVATE "${FLAVO_ENGINE_SRC_DIR}")
target_include_directories(Renderer PRIVATE "${FLAVO_RENDERER_SRC_DIR}")
target_compile_definitions(Renderer PRIVATE RENDERER_API_EXPORT)
target_link_libraries(Renderer PRIVATE RendererDX12)

set_target_properties(Renderer PROPERTIES FOLDER Renderer)
source_group(TREE ${CMAKE_CURRENT_SOURCE_DIR} FILES ${SOURCES_RENDERER})
